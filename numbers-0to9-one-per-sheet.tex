\newcommand{\snowflake}{
  \begin{tikzpicture}[scale=0.25]
    \foreach \angle in {0,60,...,300} {
      \draw[rotate=\angle, thin] 
        (0,0) -- (0,3.5)
        (0,2) -- (1,3)
        (0,2) -- (-1,3);
    }
  \end{tikzpicture}
}

}
